mass_directive_decisions_plus_submenu = {
    picture = { reference = "gfx/interface/illustrations/decisions/fp2_decision_struggle_compromise.dds" }
    desc = mass_directive_decisions_plus_submenu
    confirm_text = mass_directive_decisions_plus_submenu_confirm
    selection_tooltip = mass_directive_decisions_plus_submenu_tooltip
    decision_group_type = decisions

    is_shown = {
        any_vassal = { is_landed = yes highest_held_title_tier >= tier_county }
    }

    # ────────── Widget: Choose Sub-Governor (Admin / Celestial) ──────────
    widget = {
        gui = "decision_view_widget_decision_option_list_controller"
        controller = decision_option_list_controller

        # Only show sub-government choices if Liege is Admin or Celestial
        item = {
            value = sub_frontier
            current_description = "Frontier governors"
            localization = mass_directive_sub_frontier
            is_shown = { OR = { has_government = administrative_government has_government = celestial_government } }
            trigger_event = mass_directive_submenu.0001
            set_variable = { name = chosen_subgov value = frontier_government }
        }

        item = {
            value = sub_civil
            current_description = "Civil governors"
            localization = mass_directive_sub_civil
            is_shown = { OR = { has_government = administrative_government has_government = celestial_government } }
            trigger_event = mass_directive_submenu.0001
            set_variable = { name = chosen_subgov value = civil_government }
        }

        item = {
            value = sub_balanced
            current_description = "Balanced governors"
            localization = mass_directive_sub_balanced
            is_shown = { OR = { has_government = administrative_government has_government = celestial_government } }
            trigger_event = mass_directive_submenu.0001
            set_variable = { name = chosen_subgov value = balanced_government }
        }

        item = {
            value = sub_military
            current_description = "Military governors"
            localization = mass_directive_sub_military
            is_shown = { OR = { has_government = administrative_government has_government = celestial_government } }
            trigger_event = mass_directive_submenu.0001
            set_variable = { name = chosen_subgov value = military_government }
        }

        # Always show mass directives directly for non-Admin / Celestial Lieges
        item = {
            value = mod_improve_development
            current_description = mod_improve_development_tooltip
            localization = mod_new_decision_improve_development
            is_shown = { NOT = { OR = { has_government = administrative_government has_government = celestial_government } } }
            add_character_flag = vassal_directive_improve_development
        }

        item = {
            value = mod_train_commanders
            current_description = mod_new_train_commanders_tooltip
            localization = mod_new_decision_train_commanders
            is_shown = { NOT = { OR = { has_government = administrative_government has_government = celestial_government } } }
            add_character_flag = vassal_directive_train_commanders
        }

        # Add remaining standard directives for non-admin automatically...
        item = { value = mod_build_maa current_description = mod_build_maa_tooltip localization = mod_new_decision_build_maa is_shown = { NOT = { OR = { has_government = administrative_government has_government = celestial_government } } } add_character_flag = vassal_directive_build_maa }
        item = { value = mod_convert_faith current_description = mod_convert_faith_tooltip localization = mod_new_decision_convert_faith is_shown = { NOT = { OR = { has_government = administrative_government has_government = celestial_government } } } add_character_flag = vassal_directive_convert_faith }
        item = { value = mod_convert_culture current_description = mod_convert_culture_tooltip localization = mod_new_decision_convert_culture is_shown = { NOT = { OR = { has_government = administrative_government has_government = celestial_government } } } add_character_flag = vassal_directive_convert_culture }
        item = { value = mod_improve_cultural_acceptance current_description = mod_improve_cultural_acceptance_tooltip localization = mod_new_decision_improve_cultural_acceptance is_shown = { NOT = { OR = { has_government = administrative_government has_government = celestial_government } } } add_character_flag = vassal_directive_improve_cultural_acceptance }
        item = { value = mod_building_focus_fortification current_description = mod_building_focus_fortification_tooltip localization = mod_new_decision_building_focus_fortification is_shown = { NOT = { OR = { has_government = administrative_government has_government = celestial_government } } } add_character_flag = vassal_directive_building_focus_fortification }
        item = { value = mod_building_focus_military current_description = mod_building_focus_military_tooltip localization = mod_new_decision_building_focus_military is_shown = { NOT = { OR = { has_government = administrative_government has_government = celestial_government } } } add_character_flag = vassal_directive_building_focus_military }
        item = { value = mod_building_focus_economy current_description = mod_building_focus_economy_tooltip localization = mod_new_decision_building_focus_economy is_shown = { NOT = { OR = { has_government = administrative_government has_government = celestial_government } } } add_character_flag = vassal_directive_building_focus_economy }
        item = { value = mod_remove_directive current_description = mod_new_decision_remove_directive_tooltip localization = mod_new_decision_remove_directive is_shown = { NOT = { OR = { has_government = administrative_government has_government = celestial_government } } } }
    }

    # Effect: applies the chosen directive to vassals
    effect = {
        every_vassal = {
            limit = { is_landed = yes highest_held_title_tier >= tier_county }

            switch = {
                trigger = yes

                scope:mod_improve_development ?= {
                    limit = { OR = { has_government = administrative_government has_government = celestial_government chosen_subgov = frontier_government } }
                    remove_vassal_directives = yes
                    add_character_flag = vassal_directive_improve_development
                }

                scope:mod_train_commanders ?= {
                    limit = { OR = { has_government = administrative_government has_government = celestial_government chosen_subgov = frontier_government } }
                    remove_vassal_directives = yes
                    add_character_flag = vassal_directive_train_commanders
                }

                # Add all other directives similarly filtered by chosen_subgov...
            }
        }
    }

    # AI should never do this
    ai_will_do = { base = 0 }
    ai_check_interval = 0
}
