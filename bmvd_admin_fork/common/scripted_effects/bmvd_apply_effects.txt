########################################################################
# BMVD — Scripted Effects
# One effect per directive. Each reads bmvd_target_group variable and
# loops the correct filtered vassal set.
########################################################################

bmvd_apply_directive_train_commanders = {
    if = {
        limit = { var:bmvd_target_group = flag:all }
        every_vassal = {
            limit = { is_ruler = yes can_set_vassal_directive = { directive = train_commanders } }
            set_vassal_directive = train_commanders
        }
    }
    else_if = {
        limit = { var:bmvd_target_group = flag:admin_balanced }
        every_vassal = {
            limit = { is_ruler = yes vassal_contract_has_flag = admin_theme_balanced can_set_vassal_directive = { directive = train_commanders } }
