########################################################################
# Better Mass Vassal Directive — Admin & Celestial Fork
# Fork of Wehrmachtserdbeere's Better Mass Vassal Directive
#
# OBLIGATION CHECKS (confirmed from vanilla files):
#   Administrative government — vassal_contract_has_flag:
#     admin_theme_balanced | admin_theme_military | admin_theme_civilian
#     admin_theme_frontier | admin_theme_navy     | admin_theme_imperial
#
#   Celestial government — vassal_contract_has_flag:
#     meritocratic_civil_governor | meritocratic_military_governor
#
# FLOW:
#   Non-admin/celestial rulers  ->  direct all-vassal directive decisions
#   Admin/Celestial rulers      ->  Step 1: select group (sets variable)
#                               ->  Step 2: pick directive (fires + clears)
########################################################################

# =====================================================================
# STEP 1A  —  NON-ADMIN/CELESTIAL RULERS
# Direct decisions, one per directive, target all eligible vassals.
# =====================================================================

bmvd_mass_directive_all_train_commanders = {
    picture = "gfx/interface/illustrations/decisions/decision_dynasty_house.dds"
    desc = bmvd_mass_directive_all_train_commanders_desc
    major = yes
    ai_check_interval = 0

    is_shown = {
        is_ruler = yes
        NOT = { OR = { government = administrative_government government = celestial_government } }
        any_vassal = { is_ruler = yes }
    }
    is_valid = {
        any_vassal = { is_ruler = yes can_set_vassal_directive = { directive = train_commanders } }
    }
    effect = {
        every_vassal = {
            limit = { is_ruler = yes can_set_vassal_directive = { directive = train_commanders } }
            set_vassal_directive = train_commanders
        }
        custom_tooltip = bmvd_mass_directive_applied_tt
    }
    cost = { gold = 0 }
}

# =====================================================================
# STEP 1B  —  ADMIN/CELESTIAL: SELECT GROUP
#
# Sets variable bmvd_target_group to a flag identifying the group.
# Flags used:
#   Administrative: admin_balanced | admin_military | admin_civilian
#                   admin_frontier | admin_navy     | admin_imperial
#   Celestial:      celestial_civil | celestial_military
#   Either:         all
# =====================================================================

bmvd_admin_cancel_group = {
    picture = "gfx/interface/illustrations/decisions/decision_dynasty_house.dds"
    desc = bmvd_admin_cancel_group_desc
    major = yes
    ai_check_interval = 0

    is_shown = {
        is_ruler = yes
        OR = { government = administrative_government government = celestial_government }
        has_variable = bmvd_target_group
    }
    is_valid = { always = yes }
    effect = {
        remove_variable = bmvd_target_group
        custom_tooltip = bmvd_group_cleared_tt
    }
    cost = { gold = 0 }
}

# =====================================================================
# STEP 2  —  ADMIN/CELESTIAL: APPLY DIRECTIVE TO SELECTED GROUP
#
# One decision per directive. All shown only when bmvd_target_group is set.
# switch fans out over the 9 possible flag values from Step 1.
# After firing, bmvd_target_group is cleared automatically.
# =====================================================================

bmvd_admin_apply_train_commanders = {
    picture = "gfx/interface/illustrations/decisions/decision_dynasty_house.dds"
    desc = bmvd_admin_apply_train_commanders_desc
    major = yes
    ai_check_interval = 0

    is_shown = {
        is_ruler = yes
        OR = { government = administrative_government government = celestial_government }
        has_variable = bmvd_target_group
    }
    is_valid = {
        OR = {
            AND = { var:bmvd_target_group = flag:all                any_vassal = { is_ruler = yes can_set_vassal_directive = { directive = train_commanders } } }
            AND = { var:bmvd_target_group = flag:admin_balanced     any_vassal = { is_ruler = yes vassal_contract_has_flag = admin_theme_balanced  can_set_vassal_directive = { directive = train_commanders } } }
            AND = { var:bmvd_target_group = flag:admin_military     any_vassal = { is_ruler = yes vassal_contract_has_flag = admin_theme_military  can_set_vassal_directive = { directive = train_commanders } } }
            AND = { var:bmvd_target_group = flag:admin_civilian     any_vassal = { is_ruler = yes vassal_contract_has_flag = admin_theme_civilian  can_set_vassal_directive = { directive = train_commanders } } }
            AND = { var:bmvd_target_group = flag:admin_frontier     any_vassal = { is_ruler = yes vassal_contract_has_flag = admin_theme_frontier  can_set_vassal_directive = { directive = train_commanders } } }
            AND = { var:bmvd_target_group = flag:admin_navy         any_vassal = { is_ruler = yes vassal_contract_has_flag = admin_theme_navy      can_set_vassal_directive = { directive = train_commanders } } }
            AND = { var:bmvd_target_group = flag:admin_imperial     any_vassal = { is_ruler = yes vassal_contract_has_flag = admin_theme_imperial  can_set_vassal_directive = { directive = train_commanders } } }
            AND = { var:bmvd_target_group = flag:celestial_civil    any_vassal = { is_ruler = yes vassal_contract_has_flag = meritocratic_civil_governor    can_set_vassal_directive = { directive = train_commanders } } }
            AND = { var:bmvd_target_group = flag:celestial_military any_vassal = { is_ruler = yes vassal_contract_has_flag = meritocratic_military_governor can_set_vassal_directive = { directive = train_commanders } } }
        }
    }
    effect = {
        switch = {
            trigger = var:bmvd_target_group
            flag:all                = { every_vassal = { limit = { is_ruler = yes can_set_vassal_directive = { directive = train_commanders } }                                                                         set_vassal_directive = train_commanders } }
            flag:admin_balanced     = { every_vassal = { limit = { is_ruler = yes vassal_contract_has_flag = admin_theme_balanced  can_set_vassal_directive = { directive = train_commanders } }                        set_vassal_directive = train_commanders } }
            flag:admin_military     = { every_vassal = { limit = { is_ruler = yes vassal_contract_has_flag = admin_theme_military  can_set_vassal_directive = { directive = train_commanders } }                        set_vassal_directive = train_commanders } }
            flag:admin_civilian     = { every_vassal = { limit = { is_ruler = yes vassal_contract_has_flag = admin_theme_civilian  can_set_vassal_directive = { directive = train_commanders } }                        set_vassal_directive = train_commanders } }
            flag:admin_frontier     = { every_vassal = { limit = { is_ruler = yes vassal_contract_has_flag = admin_theme_frontier  can_set_vassal_directive = { directive = train_commanders } }                        set_vassal_directive = train_commanders } }
            flag:admin_navy         = { every_vassal = { limit = { is_ruler = yes vassal_contract_has_flag = admin_theme_navy      can_set_vassal_directive = { directive = train_commanders } }                        set_vassal_directive = train_commanders } }
            flag:admin_imperial     = { every_vassal = { limit = { is_ruler = yes vassal_contract_has_flag = admin_theme_imperial  can_set_vassal_directive = { directive = train_commanders } }                        set_vassal_directive = train_commanders } }
            flag:celestial_civil    = { every_vassal = { limit = { is_ruler = yes vassal_contract_has_flag = meritocratic_civil_governor    can_set_vassal_directive = { directive = train_commanders } }               set_vassal_directive = train_commanders } }
            flag:celestial_military = { every_vassal = { limit = { is_ruler = yes vassal_contract_has_flag = meritocratic_military_governor can_set_vassal_directive = { directive = train_commanders } }               set_vassal_directive = train_commanders } }
        }
        remove_variable = bmvd_target_group
        custom_tooltip = bmvd_mass_directive_applied_tt
    }
    cost = { gold = 0 }
}
